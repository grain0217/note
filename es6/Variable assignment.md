## 变量的解析赋值

#### 解构

ES6中可以这样赋值：

    let [a,b,c] = ["q","w","e"];
按照对应模式赋值：

    let [a,,c] = ["q","w","e"];
    a//"q"
    c//"e"

    let [x,...y] = [1,2,3,4]
    x//1
    y//[2,3,4]

    let [m,n] = [1]
    m//1
    n// undefined

嵌套赋值：

    let [a,[[b],c]] = [1,[[2,3,4],[6,7]]
    a//1
    b//2
    c//[6,7]
#### '='右边为不可遍历的情况

    let [x,y] = 1
    let [x,y] = {}
    都会报错

#### 默认值

    let [x,y=2] = [1]
    x//1
    y//2

    let [a,b='b'] = ['a', undefined]
    a//'a'
    b//'b'
