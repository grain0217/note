#### 函数防抖

	const debounce = (fn, delay) => {
		let timer = null;
		return function () {
		    let context = this, args = arguments;
		    clearTimeout(timer);
		    timer = setTimeout(function () {
				fn.apply(context, args);
		    }, delay);
		};
	}

	input.onkeypress = debounce(function(){
		// ajax request
	}， 500）;
