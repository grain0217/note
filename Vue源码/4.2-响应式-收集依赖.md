前面我们讲到`defineReactive`通过`Object.defineProperty`为被监测的数据添加`getter`：
```js
function defineReactive (obj, key, val, customSetter, shallow) {
  const dep = new Dep()

  const property = Object.getOwnPropertyDescriptor(obj, key)
  if (property && property.configurable === false) return

  const getter = property && property.get
  const setter = property && property.set
  if ((!getter || setter) && arguments.length === 2) {
    val = obj[key]
  }

  let childOb = !shallow && observe(val)

  Object.defineProperty(obj, key, {
    enumerable: true,
    configurable: true,
    get: function reactiveGetter () {
      const value = getter ? getter.call(obj) : val
      if (Dep.target) {
        dep.depend()
        if (childOb) {
          childOb.dep.depend()
          if (Array.isArray(value)) {
            dependArray(value)
          }
        }
      }
      return value
    }
  })
}
```

`getter`中有两个需要关注的地方：
- 一个是实例化了一个`Dep`的实例
- 另一个是在观测数据对象的`key`属性的`getter`中有一个`dep.depend`的操作，它是用来做依赖收集的

### Dep的定义
```js
class Dep {
  constructor () {
    this.id = uid++
    this.subs = []
  }

  addSub () {}

  notify () {}

const targetStack = []
```

`Dep`是通过对`Watcher`的管理实现依赖收集，所以`Dep`实例需要通过一个`subs`数组维护它所管理的`Watcher`实例。此外`Dep`类还有一个静态属性`target`，这是一个**唯一的全局**`Watcher`。

`Dep`原型上还有一些依赖收集相关的`addSub`、`removeSub`、`depend`、`notify`方法，在了解它们之前，需要先来看一看什么是`Watcher`。

### Wather的定义
```js
class Watcher {
  constructor (vm, expOrFn, cb, options, isRenderWatcher) {
    this.vm = vm
    if (isRenderWatcher) vm._watcher = this

    vm._watchers.push(this)

      this.deep = !!options.deep
      this.user = !!options.user
      this.lazy = !!options.lazy
      this.sync = !!options.sync
      this.before = options.before
    } else {
      this.deep = this.user = this.lazy = this.sync = false
    }
    this.cb = cb
    this.id = ++uid

    this.dirty = this.lazy
    this.deps = []
    this.newDeps = []
    this.depIds = new Set()
    this.newDepIds = new Set()
    this.expression = process.env.NODE_ENV !== 'production' ? expOrFn.toString() : ''
      
    if (typeof expOrFn === 'function') {
      this.getter = expOrFn
    } else {
      this.getter = parsePath(expOrFn)
    }
    this.value = this.lazy ? undefined : this.get()
  }

  get () {}

  addDep () {}

  cleanupDeps () {}

  // ... 派发依赖相关的方法
  update () {}

  run () {}

  evaluate () {}

  depend () {}

  teardown () {}
}
```

`Watcher`的构造函数中定义了一些和`Dep`相关的属性——`deps`、`newDeps`、`depIds`、`newDepIds`。其中`deps`和`newDeps`是用来维护`Dep`实例的数组，而`depIds`和`newDepIds`是`Set`类型结构，分别存储的是`deps`和`newDeps`中`Watcher`实例的`id`。

此外，`Watcher`中还有一些和依赖收集相关的`get`、`addDep`和`cleanupDeps`方法，下面我们在分析`Watcher`和`Dep`是如何实现依赖收集的过程中认识这些方法。

### 依赖收集的过程
我们知道，对数据对象的访问会触发它们的`getter`方法，问题是，这些响应式对象是什么时候被访问的呢？在分析Vue实例的`$mount`过程时曾讲过`mountComponent`方法：
```js
updateComponent = () => {
  vm._update(vm._render(), hydrating)
}
new Watcher(vm, updateComponent, noop, {
  before () {
    if (vm._isMounted) {
      callHook(vm, 'beforeUpdate')
    }
  }
}, true)
```

在上面对`Watcher`类的分析中可得，在初始化一个`Watcher`实例时：`this.value = this.lazy ? undefined : this.get()`会执行实例的`get`方法：
```js
// observer/watcher.js
class Watcher {
  constructor (vm, expOrFn) {
    if (typeof expOrFn === 'function') {
      this.getter = expOrFn
    } else {
      this.getter = parsePath(expOrFn)
    }
    this.value = this.lazy ? undefined : this.get()
  }

  get () {
    pushTarget(this)
    let value
    const vm = this.vm
    try {
      value = this.getter.call(vm, vm)
    } catch (e) {
      if (this.user) {
        handleError(e, vm, `getter for watcher "${this.expression}"`)
      } else {
        throw e
      }
    } finally {
      if (this.deep) {
        traverse(value)
      }
      popTarget()
      this.cleanupDeps()
    }
    return value
  }
}

// observer/dep.js
Dep.target = null
const targetStack = []

function pushTarget (target) {
  targetStack.push(target)
  Dep.target = target
}

function popTarget () {
  targetStack.pop()
  Dep.target = targetStack[targetStack.length - 1]
}
```

在`Watcher`实例`get`方法的执行流程中，首先是`pushTarget(this)`，这一步将当前`Watcher`实例：
1. 赋值给`Dep.target`
2. 推入全局变量`targetStack`数组中

紧接着是`value = this.getter.call(vm, vm)`，这里`getter`方法对应的正是`vm._update(vm._render(), hydrating)`，而在`vm.render()`**生成vnode的过程中涉及到了对实例上的数据的访问**，现在前面的问题也就有了答案——实例化Watcher时触发了响应式对象属性的`getter`。现在再回头来看一下`defineReactive`在响应式对象的属性上添加的`getter`：
```js
function defineReactive (obj, key, val, customSetter, shallow) {
  const dep = new Dep()

  val = obj[key]
  Object.defineProperty(obj, key, {
    enumerable: true,
    configurable: true,
    get: function reactiveGetter () {
      const value = val
      if (Dep.target) {
        dep.depend()
        if (childOb) {
          childOb.dep.depend()
          if (Array.isArray(value)) {
            dependArray(value)
          }
        }
      }
      return value
    }
  })
}
```

可以看到，对象属性在被处理为响应式的时候都会首先创建一个`Dep`实例，然后在属性的`getter`被触发时这个`Dep`实例的`depend()`方法得到执行，再次回到`Dep`中：
```js
class Dep {
  constructor () {
    this.id = uid++
    this.subs = []
  }
  depend () {
    if (Dep.target) {
      Dep.target.addDep(this)
    }
  }
}
```

刚才已经提到，初始化`Watcher`时已经将`Dep.target`指向了创建的`Watcher`实例，下面来看`Watcher`的`addDep`：
```js
class Watcher {
  constructor () {}

  addDep (dep) {
    const id = dep.id
    if (!this.newDepIds.has(id)) {
      this.newDepIds.add(id)
      this.newDeps.push(dep)
      if (!this.depIds.has(id)) {
        dep.addSub(this)
      }
    }
  }
}
```

这里在将`Dep`实例记录进`Watcher`实例`newDepIds`、`newDeps`之前，先做了一些保证`Dep`实例唯一性的逻辑判断，紧接着又调用了`dep.addSub(this)`：
```js
class Dep {
  constructor () {
    this.id = uid++
    this.subs = []
  }
  addSub (sub) {
    this.subs.push(sub)
  }
}
```

这里将当前的`Watcher`实例添加到这个响应式数据内的`dep.subs`中，为后续数据变化时通知这些`subs`做准备。

### 总结
这一小节对Vue响应式数据的依赖收集过程的一些细节做了分析，收集依赖的目的是在这些响应式数据发生变化触发它们的`setter`时方便通知到对应的订阅者`Watcher`去做相应的逻辑处理，这个过程叫派发更新。
